Examples

Grid allows you to create custom layouts using combinations of gridTemplate and gridArea props.

Header
Sidebar
Content
Footer

Edit the code below to see your changes live!

function Example() {
  const template = `
    "head head" 80px
    "nav  main" 200px
    "foot  foot" 50px
    / 1fr 5fr;
  `;
  return (
    <Grid gridTemplate={template}>
      <GridItem gridArea="head">
        <ExampleBox>Header</ExampleBox>
      </GridItem>
      <GridItem gridArea="nav">
        <ExampleBox>Sidebar</ExampleBox>
      </GridItem>
      <GridItem gridArea="main">
        <ExampleBox>Content</ExampleBox>
      </GridItem>
      <GridItem gridArea="foot">
        <ExampleBox>Footer</ExampleBox>
      </GridItem>
    </Grid>
  );
}
